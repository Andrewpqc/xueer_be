## 0. build
### virtual environment

    $ virtual venv
    $ source venv/bin/activate
    $ pip install -r requirement.text (--no-cache-dir)
    
### test database

    $ chmod 777 data.sh
    $ ./data.sh
    >> Role.insert_roles()
    >> quit()
    
### add user

    $ python manage.py adduser username
